Data Preprocessing

# Handle missing values
df <- na.omit(df)

# Convert categorical variables to factors
df$trustLevel <- as.factor(df$trustLevel)
df$fraud <- as.factor(df$fraud)

# Normalize numerical features
normalize <- function(x) {
  return((x - min(x)) / (max(x) - min(x)))
}

df[, c("totalScanTimeInSeconds", "grandTotal", "scannedLineItemsPerSecond", "valuePerSecond", "lineItemVoidsPerPosition")] <- 
  lapply(df[, c("totalScanTimeInSeconds", "grandTotal", "scannedLineItemsPerSecond", "valuePerSecond", "lineItemVoidsPerPosition")], normalize)

Train-Test Split

set.seed(42)
trainIndex <- createDataPartition(df$fraud, p = 0.8, list = FALSE)
df_train <- df[trainIndex, ]
df_test <- df[-trainIndex, ]

XGBoost

# Ensure all features are numeric
df_train_numeric <- df_train %>% select(-fraud) %>% mutate(across(everything(), as.numeric))
df_test_numeric <- df_test %>% select(-fraud) %>% mutate(across(everything(), as.numeric))

# Convert target variable to numeric (XGBoost requires 0/1)
train_labels <- as.numeric(df_train$fraud) - 1
test_labels <- as.numeric(df_test$fraud) - 1

# Create XGBoost matrices
train_matrix <- xgb.DMatrix(data = as.matrix(df_train_numeric), label = train_labels)
test_matrix <- xgb.DMatrix(data = as.matrix(df_test_numeric), label = test_labels)

# Train the XGBoost model
xgb_model <- xgboost(
  data = train_matrix,
  nrounds = 50,
  objective = "binary:logistic"
)
